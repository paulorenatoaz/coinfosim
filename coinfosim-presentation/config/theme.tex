% Paleta principal.
\definecolor{CoInfoInk}{HTML}{3B3E58}      % texto principal: azul-ameixa profundo
\definecolor{CoInfoBlue}{HTML}{3F68B0}     % acento primário (série 1 dos gráficos)
\definecolor{CoInfoAmber}{HTML}{B98136}    % acento quente (série 2 dos gráficos)
\definecolor{CoInfoPlum}{HTML}{964F9C}     % acento ameixa (série 3 dos gráficos)
\definecolor{CoInfoMint}{HTML}{4E8A78}     % verde-menta contido (afirmações positivas)
\definecolor{CoInfoRose}{HTML}{B0566A}     % rosa queimado contido (negações/limites)
\definecolor{CoInfoLavender}{HTML}{8B84C0} % lavanda para detalhes decorativos discretos
\definecolor{CoInfoGray}{HTML}{6C6980}     % texto secundário
\definecolor{CoInfoPale}{HTML}{EFF1F9}     % painel azul-lavanda pálido
\definecolor{CoInfoCream}{HTML}{FAF5EB}    % painel creme quente (takeaways)
\definecolor{CoInfoMist}{HTML}{F6F5FA}     % fundo neutro claríssimo
\definecolor{CoInfoLine}{HTML}{D9D7E4}     % fios e divisórias suaves

% Aliases para compatibilidade com nomes antigos.
